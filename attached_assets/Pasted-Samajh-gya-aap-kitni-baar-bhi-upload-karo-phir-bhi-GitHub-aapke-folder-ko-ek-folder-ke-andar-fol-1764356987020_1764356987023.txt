import * as FileSystem from 'expo-file-system'; import axios from 'axios'; const uploadToGitHub = async (folderPath, repo, branch, token) => { const api = axios.create({ baseURL: `https://api.github.com/repos/${repo}/contents/`, headers: { Authorization: `Bearer ${token}`, "Content-Type": "application/json" } }); // STEP 1: R√©cursive function (reads files inside folder) const processFolder = async (currentPath, rootPath) => { const items = await FileSystem.readDirectoryAsync(currentPath); for (const item of items) { const itemFullPath = `${currentPath}/${item}`; const stats = await FileSystem.getInfoAsync(itemFullPath); if (stats.isDirectory) { await processFolder(itemFullPath, rootPath); // read inner folders } else { const fileContent = await FileSystem.readAsStringAsync(itemFullPath, { encoding: 'base64' }); // IMPORTANT PART: remove root folder name from path const relativePath = itemFullPath.replace(rootPath + '/', ''); // upload each file WITHOUT root folder await api.put(relativePath, { message: `upload: ${relativePath}`, content: fileContent, branch: branch }); console.log("Uploaded:", relativePath); } } }; // STEP 2: Start from root folder await processFolder(folderPath, folderPath); };
